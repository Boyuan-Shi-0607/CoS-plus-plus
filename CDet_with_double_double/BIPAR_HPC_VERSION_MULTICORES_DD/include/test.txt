void run_tests_fpm_summed() {

    // particularly with singular mats
    std::cout << "\n=== FPM Computer Correctness Tests (DoubleDouble) ===" << std::endl;
    std::mt19937 rng(1232454);

    std::cout << " n |           Max Error | Status" << std::endl;
    std::cout << "---|---------------------|--------" << std::endl;
    for (int n = 2; n <= 8; ++n) {
        int N = 2 * n;
        std::vector<DoubleDouble> A_up(N * N), A_down(N * N, DoubleDouble(0.0));
        std::uniform_real_distribution<double> val_dist(-5.0, 5.0);
        for (int i = 0; i < N * N; ++i) {
            if (rng() % 5 != 0) A_up[i] = DoubleDouble(val_dist(rng));
            else A_up[i] = DoubleDouble(0.0);
        }
        for (int i = 0; i < N; ++i) A_down[i*N + i] = DoubleDouble(1.0);

        FPMComputer<DoubleDouble> computer;
        auto result_fpm = computer.compute_postprocessed_minors(A_up, A_down, n);

        std::vector<DoubleDouble> result_brute;
        brute_force_postprocessed(A_up, A_down, n, result_brute);

        double max_error = 0.0;
        for(size_t i = 0; i < result_fpm.size(); ++i) {
            max_error = std::max(max_error, MathOps<DoubleDouble>::abs(result_fpm[i] - result_brute[i]).to_double());
        }
        bool passed = max_error < 1e-25;

        std::cout << std::setw(2) << n << " | "
                  << std::setw(19) << std::scientific << std::setprecision(3) << max_error << " | "
                  << (passed ? "PASS" : "FAIL") << std::endl;
    }
}

// Test function to verify sublattice summing
void test_sublattice_summing() {
    std::cout << "\n=== Testing Sublattice Summing ===" << std::endl;

    /* Test Logic:
     * 1. Generate random args of size 3*n (for tau and R values)
     * 2. Original method: Extend to 4*n by adding n sublattice indices
     *    - Keep the same first 3*n values
     *    - Vary the last n values to cover all 2^n sublattice configurations
     *    - Sum all 2^n evaluations
     * 3. Summed method: Pass only 3*n args to sublattice-summed function
     *    - This internally handles all 2^n configurations
     * 4. Compare results - they should match
     *
     * Example for n=2:
     * - args_3n = [tau1, R1_1, R1_2, tau2, R2_1, R2_2]  (6 values)
     * - args_4n = [tau1, R1_1, R1_2, tau2, R2_1, R2_2, sub1, sub2]  (8 values)
     * - We sum over 4 configurations:
     *   config 0: sub1=0, sub2=0
     *   config 1: sub1=1, sub2=0
     *   config 2: sub1=0, sub2=1
     *   config 3: sub1=1, sub2=1
     */

    // Test parameters
    const int order_num = 9;  // You can change this
    const int i = 0;  // Grid index
    const int j = 0;  // Beta index

    // First, let's demonstrate the concept with a simple example
    std::cout << "\nDemonstration with order_num = 2:" << std::endl;
    std::cout << "We have args_3n = [tau1, R1_b1, R1_b2, tau2, R2_b1, R2_b2]" << std::endl;
    std::cout << "We extend to args_4n = [tau1, R1_b1, R1_b2, tau2, R2_b1, R2_b2, sub1, sub2]" << std::endl;
    std::cout << "Then sum over 2^2 = 4 configurations of (sub1, sub2):" << std::endl;
    std::cout << "  (0,0), (1,0), (0,1), (1,1)" << std::endl;
    std::cout << "The sublattice-summed function does this sum internally.\n" << std::endl;

    // Random number generator
    std::random_device rd;
    std::mt19937 gen(rd());
    std::uniform_real_distribution<double> dist(0.0, 1.0);

    // Generate random args of size 3*order_num (these stay fixed for the comparison)
    std::vector<DoubleDouble> args_3n(3 * order_num);
    for (int k = 0; k < 3 * order_num; ++k) {
        args_3n[k] = DoubleDouble(dist(gen));
    }

    std::cout << "Testing with order_num = " << order_num << std::endl;
    std::cout << "Grid index i = " << i << ", Beta index j = " << j << std::endl;
    std::cout << "Fixed prefix: args[0.." << 3*order_num-1 << "] (tau and R values)" << std::endl;

    // Method 1: Sum over all sublattice configurations using original integrand
    auto sum_original = DoubleDouble(0);
    std::vector<DoubleDouble> args_4n(4 * order_num);

    // IMPORTANT: Keep the same prefix - copy first 3*order_num args
    // This ensures we're testing with the SAME tau and R values
    for (int k = 0; k < 3 * order_num; ++k) {
        args_4n[k] = args_3n[k];  // Same prefix for all configurations
    }

    // Time the original method
    auto start_original = std::chrono::high_resolution_clock::now();

    // Loop over all 2^order_num sublattice configurations
    // This spans all possible combinations of sublattice indices
    int num_configs = 1 << order_num;  // 2^order_num configurations

    std::cout << "\nSumming over " << num_configs << " sublattice configurations:" << std::endl;
    std::cout << "Example for order_num=" << order_num << ":" << std::endl;
    std::cout << "  args_4n[0.." << 3*order_num-1 << "] = args from args_3n (tau's and R's)" << std::endl;
    std::cout << "  args_4n[" << 3*order_num << ".." << 4*order_num-1 << "] = sublattice indices" << std::endl;

    for (int config = 0; config < num_configs; ++config) {
        // Set sublattice indices for positions 3n to 4n-1 based on bits of config
        // Each bit determines sublattice 0 or 1 for that position
        for (int k = 0; k < order_num; ++k) {
            // The original integrand extracts sublattice as: floor(2 * args[3*n + k])
            // So we use:
            // - 0.1 for sublattice 0: floor(2 * 0.1) = floor(0.2) = 0
            // - 0.7 for sublattice 1: floor(2 * 0.7) = floor(1.4) = 1
            args_4n[3 * order_num + k] = ((config >> k) & 1) ? DoubleDouble(0.7) : DoubleDouble(0.1);
        }

        // Debug: print first few configurations
        if (config < 4 || config == num_configs - 1) {
            std::cout << "  Config " << std::setw(2) << config << " (binary: ";
            for (int b = order_num - 1; b >= 0; --b) {
                std::cout << ((config >> b) & 1);
            }
            std::cout << "): sublattices = ";
            for (int k = 0; k < order_num; ++k) {
                int sublattice = DD_FLOOR_TO_INT(DoubleDouble(2) * args_4n[3 * order_num + k]);
                std::cout << sublattice << " ";
            }
            if (config == 3 && num_configs > 4) std::cout << "\n  ...";
            std::cout << std::endl;
        }

        // Evaluate original integrand with this sublattice configuration
        DoubleDouble val = integrand_dd(args_4n.data(), order_num, i, j);
        sum_original = sum_original + val;
    }

    auto end_original = std::chrono::high_resolution_clock::now();
    double time_original = std::chrono::duration<double, std::milli>(end_original - start_original).count();

    // Method 2: Use sublattice-summed integrand
    auto start_summed = std::chrono::high_resolution_clock::now();
    DoubleDouble result_summed = integrand_dd_sublattice_summed(args_3n.data(), order_num, i, j);
    auto end_summed = std::chrono::high_resolution_clock::now();
    double time_summed = std::chrono::duration<double, std::milli>(end_summed - start_summed).count();

    // Compare results
    DoubleDouble diff = DD_ABS(sum_original - result_summed);
    DoubleDouble rel_diff = diff / DD_ABS(sum_original);

    std::cout << std::fixed << std::setprecision(30);
    std::cout << "\nResults:" << std::endl;
    std::cout << "Original sum over " << num_configs << " configs: " << sum_original.to_double() << std::endl;
    std::cout << "  = Sum of integrand_dd evaluated at all " << num_configs << " sublattice configurations" << std::endl;
    std::cout << "Sublattice-summed result:                  " << result_summed.to_double() << std::endl;
    std::cout << "  = Single call to integrand_sublattice_summed_dd" << std::endl;
    std::cout << "Absolute difference: " << diff.to_double() << std::endl;
    std::cout << "Relative difference: " << rel_diff.to_double() << std::endl;

    std::cout << "\nTiming:" << std::endl;
    std::cout << "Original method: " << time_original << " ms" << std::endl;
    std::cout << "Summed method:   " << time_summed << " ms" << std::endl;
    std::cout << "Speedup: " << time_original / time_summed << "x" << std::endl;

    // Test if results match within tolerance
    auto tolerance = DoubleDouble(1e-20);
    if (rel_diff < tolerance) {
        std::cout << "\n✅ TEST PASSED: Results match within tolerance" << std::endl;
    } else {
        std::cout << "\n❌ TEST FAILED: Results differ by more than tolerance" << std::endl;
    }

    // Additional timing test with multiple evaluations
    std::cout << "\n--- Performance Test (100 evaluations) ---" << std::endl;

    const int num_tests = 100;
    auto dummy_sum = DoubleDouble(0);

    // Time original method
    start_original = std::chrono::high_resolution_clock::now();
    for (int test = 0; test < num_tests; ++test) {
        // Generate new random args for first 3n positions
        for (int k = 0; k < 3 * order_num; ++k) {
            args_3n[k] = DoubleDouble(dist(gen));
            args_4n[k] = args_3n[k];  // Keep same prefix
        }

        // Sum over all 2^n sublattice configurations
        for (int config = 0; config < num_configs; ++config) {
            // Set sublattice indices for last n positions
            for (int k = 0; k < order_num; ++k) {
                args_4n[3 * order_num + k] = ((config >> k) & 1) ? DoubleDouble(0.7) : DoubleDouble(0.1);
            }
            dummy_sum = dummy_sum + integrand_dd(args_4n.data(), order_num, i, j);
        }
    }
    end_original = std::chrono::high_resolution_clock::now();
    time_original = std::chrono::duration<double, std::milli>(end_original - start_original).count();

    // Time summed method
    start_summed = std::chrono::high_resolution_clock::now();
    for (int test = 0; test < num_tests; ++test) {
        // Generate new random args
        for (int k = 0; k < 3 * order_num; ++k) {
            args_3n[k] = DoubleDouble(dist(gen));
        }

        dummy_sum = dummy_sum + integrand_dd_sublattice_summed(args_3n.data(), order_num, i, j);
    }
    end_summed = std::chrono::high_resolution_clock::now();
    time_summed = std::chrono::duration<double, std::milli>(end_summed - start_summed).count();

    std::cout << "Original method (" << num_tests << " × " << num_configs << " evaluations): "
              << time_original << " ms" << std::endl;
    std::cout << "Summed method (" << num_tests << " evaluations):                "
              << time_summed << " ms" << std::endl;
    std::cout << "Speedup: " << time_original / time_summed << "x" << std::endl;
    std::cout << "Average time per evaluation (summed): " << time_summed / num_tests << " ms" << std::endl;
}
